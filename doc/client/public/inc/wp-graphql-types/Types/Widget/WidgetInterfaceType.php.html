          'widgetId'          => [
            'type'    => Types::string(),

            },
          ],
          'name'          => [
            'type'    => Types::string(),
            'resolve' => function( array $widget, $args, AppContext $context, ResolveInfo $info ) {
              return ( ! empty( $widget[ 'name' ] ) ) ? $widget[ 'name' ] : '';
            },
          ],
          'basename'          => [
            'type'    => Types::string(),
