	public static function root_query() {
		if ( null === self::$root_query ) {
			self::$root_query = [
				'type'        => CustomTypes::theme_mods(),
				'resolve'     => function () {
					return ExtraSource::resolve_theme_mods_data();
				},
			];
		}

		return self::$root_query;
	}
