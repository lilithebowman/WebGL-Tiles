    private static function update_theme_mods( $theme_mods_args ) {

      if( is_array( $theme_mods_args ) && ! empty( $theme_mods_args ) ) {
        foreach( $theme_mods_args as $name => $value ) {
          set_theme_mod( $name, $value );
        }
        return true;
      }

      return false;

    }
